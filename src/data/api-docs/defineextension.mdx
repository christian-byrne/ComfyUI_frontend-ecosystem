Register an app-wide extension with lifecycle hooks and shell UI contributions.
Use this for global initialization, registering commands, sidebar tabs, and bottom panels.

## Type Signature

```ts
interface ExtensionOptions {
  /** Globally unique extension name, e.g. 'my-org.my-extension' */
  name: string

  /**
   * Runs once during app initialization (after mount, before first workflow).
   * Equivalent to v1 `ComfyExtension.init`.
   */
  init?(): void | Promise<void>

  /**
   * Runs once after app and all core extensions are initialized.
   * Safe to call shell UI registration APIs here.
   * Equivalent to v1 `ComfyExtension.setup`.
   */
  setup?(): void | Promise<void>
}
```

## Usage Example

```ts
import { defineExtension } from '@comfyorg/extension-api'

export default defineExtension({
  name: 'my-org.my-extension',

  // Early initialization — app mounted but no workflow yet
  async init() {
    console.log('Extension initializing...')
  },

  // App ready — register commands, sidebar tabs, bottom panels
  async setup() {
    // Register commands via CommandManager
    // Add sidebar tabs via ExtensionManager
    // Configure bottom panels
    console.log('Extension setup complete')
  }
})
```
